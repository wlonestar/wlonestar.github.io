自己写的简单Java矩阵库

最近正好在学线性代数，就想着能不能用Java写个计算矩阵运算的类库，花了几个小时，简单实现了一个计算向量点乘、矩阵乘法、转置的库。

突然想到行列式的值还没写进去，等有时间再说吧。

/**
 * a class solves Matrix multiply and transpose
 * @version 1.0
 * @author wjl
 * */
public class Matrix {
    /**
     * @param x a vector
     * @param y a vector
     * */
    private double[] x;
    private double[] y;

    /**
     * @param a a matrix
     * @param b a matrix
     * */
    private double[][] a;
    private double[][] b;

    public double[] getX() {
        return x;
    }

    public double[] getY() {
        return y;
    }

    public double[][] getA() {
        return a;
    }

    public double[][] getB() {
        return b;
    }

    public void setB(double[][] b) {
        this.b = b;
    }

    /**
     * Dot Multiplication
     * */
    public static double dot(double[] x, double[] y) {
        double result = 0.0;
        for (int i = 0; i < x.length; i++) {
            result += x[i] * y[i];
        }
        return result;
    }

    /**
     * Matrix multiplication
     * */
    public static double[][] mult(double[][] a, double[][] b) {
        double[][] c = new double[a.length][b[0].length];
        for (int i = 0; i < a.length; i++) {
            for (int j = 0; j < b[0].length; j++) {
                c[i][j] = 0;
                for (int x = 0; x < a[0].length; x++) {
                        c[i][j] += a[i][x] * b[x][j];
                }
            }
        }
        return c;
    }

    /**
     * Matrix transpose
     * */
    public static double[][] transpose(double[][] a) {
        double[][] aT = new double[a[0].length][a.length];
        for (int i = 0; i < a[0].length; i++) {
            for (int j = 0; j < a.length; j++) {
                aT[i][j] = a[j][i];
            }
        }
        return aT;
    }

    /**
     * Matrix vector multiplication
     * */
    public static double[] mult(double[][] a, double[] x) {
        double[] aX = new double[a.length];
        for (int i = 0; i < a.length; i++) {
            aX[i] = 0;
            for (int j = 0; j < a[0].length; j++) {
                aX[i] += a[i][j] * x[i];
            }
        }
        return aX;
    }

    /**
     * print vector
     * */
    public static void print(double[] x) {
        System.out.println("[" + x[0]);
        for (int i = 1; i < x.length - 1; i++) {
            System.out.println(" " + x[i]);
        }
        System.out.println(" " + x[x.length - 1] + "]");
    }

    /**
     * print matrix
     * */
    public static void print(double[][] a) {
        System.out.print("[");
        System.out.print("[");
        for (int i = 0; i < a[0].length - 1; i++)
            System.out.print(" " + a[0][i]);
        System.out.println(" " + a[0][a[0].length - 1] + " ]");
        for (int i = 1; i < a.length - 1; i++) {
            System.out.print(" [");
            for (int j = 0; j < a[0].length - 1; j++)
                System.out.print(" " + a[i][j]);
            System.out.println(" " + a[i][a[0].length - 1] + " ]");
        }
        System.out.print(" [");
        for (int i = 0; i < a[0].length - 1; i++)
            System.out.print(" " + a[a.length - 1][i]);
        System.out.println(" " + a[a.length - 1][a[0].length - 1] + " ]]");
    }
}
