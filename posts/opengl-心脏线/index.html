OpenGL 心脏线

2022-04-18
3分钟阅读时长
学习笔记

最近在学 OpenGL ，但是 OpenGL 只支持点、线、三角形，所以想着把基本图形的绘制封装成头文件，方便后续调用

心脏线类

这里是绘制心脏线的头文件，尽量使用面向对象的方法进行封装，绘制时只需要调用 draw() 函数即可

// heart.h
//
// Created by wjl15 on 2022/4/18.
//

#ifndef OPENGL_PROJECT_HEART_H
#define OPENGL_PROJECT_HEART_H

#pragma once

#include <glwjl/manager.h>
#include <cmath>

#define T 100

class heart {
  glm::vec3 center;
  float radius;

  float *_vertices{};
  unsigned int *_indices{};

  void init();

public:
  heart(glm::vec3 center, float radius);
  void draw();
};

void heart::init() {
  int num = T;
  _vertices = new float[6 * num];
  for (int i = 0; i < num; i++) {
    float angle = (2 * 3.1415f) * (i / (float) num);
    _vertices[i * 6 + 0] = 2 * radius * cosf(angle) * (1 - cosf(angle)) + center.x;
    _vertices[i * 6 + 1] = 2 * radius * sinf(angle) * (1 - cosf(angle)) + center.y;
    _vertices[i * 6 + 2] = 0.0f;

    _vertices[i * 6 + 3] = 1.0f;
    _vertices[i * 6 + 4] = 0.4f;
    _vertices[i * 6 + 5] = 0.2f;
  }
  _indices = new unsigned int[2 * num];
  for (int i = 0; i < num - 1; i++) {
    _indices[i * 2 + 0] = i;
    _indices[i * 2 + 1] = i + 1;
  }
  _indices[(num - 1) * 2 + 0] = num - 1;
  _indices[(num - 1) * 2 + 1] = 0;
}

heart::heart(glm::vec3 center, float radius) : center(center), radius(radius) {
  init();
}

void heart::draw() {
  shader shader = manager::load_shader("shader/line.vert", "shader/line.frag", nullptr, "line");

  float vertices[6 * T];
  unsigned int indices[2 * T];
  for (int i = 0; i < 6 * T; i++) {
    vertices[i] = _vertices[i];
  }
  for (int i = 0; i < 2 * T; i++) {
    indices[i] = _indices[i];
  }

  unsigned int VBO, VAO;
  glGenVertexArrays(1, &VAO);
  glGenBuffers(1, &VBO);
  glBindVertexArray(VAO);

  glBindBuffer(GL_ARRAY_BUFFER, VBO);
  glBufferData(GL_ARRAY_BUFFER, sizeof(vertices), vertices, GL_STATIC_DRAW);
  glVertexAttribPointer(0, 3, GL_FLOAT, GL_FALSE, 6 * sizeof(float), (void *) nullptr);
  glEnableVertexAttribArray(0);
  glVertexAttribPointer(1, 3, GL_FLOAT, GL_FALSE, 6 * sizeof(float), (void *) (3 * sizeof(float)));
  glEnableVertexAttribArray(1);

  shader.use();
  glBindVertexArray(VAO);
  glDrawArrays(GL_LINE_STRIP, 0, T);
  glBindVertexArray(0);
}

#endif//OPENGL_PROJECT_HEART_H

测试程序

这里是测试程序

为了保持代码整洁，进行了多次封装，具体代码可查看 LearnOpenGL 官网

// draw_heart.cpp
//
// Created by wjl15 on 2022/4/18.
//

#include <glad/glad.h>
#include <GLFW/glfw3.h>
#include <iostream>
#include <glwjl/manager.h>

#include "util/heart.h"

void framebuffer_size_callback(GLFWwindow *window, int width, int height);
void process_input(GLFWwindow *window);

int main() {
  glfwInit();
  glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR, 3);
  glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR, 3);
  glfwWindowHint(GLFW_OPENGL_PROFILE, GLFW_OPENGL_CORE_PROFILE);
  GLFWwindow *window = glfwCreateWindow(800, 800, "LearnOpenGL", nullptr, nullptr);
  if (window == nullptr) {
    std::cout << "Failed to get_value GLFW window\n";
    glfwTerminate();
    return -1;
  }
  glfwMakeContextCurrent(window);
  glfwSetFramebufferSizeCallback(window, framebuffer_size_callback);
  if (!gladLoadGLLoader((GLADloadproc) glfwGetProcAddress)) {
    std::cout << "Failed to initialize GLAD\n";
    return -1;
  }

  heart heart(glm::vec3(0.6f,0.0f,0.0f), 0.35f);

  while (!glfwWindowShouldClose(window)) {
    process_input(window);
    glClearColor(0.2f, 0.3f, 0.3f, 1.0f);
    glClear(GL_COLOR_BUFFER_BIT);

    heart.draw();

    glfwSwapBuffers(window);
    glfwPollEvents();
  }

  glfwTerminate();
  return 0;
}

void framebuffer_size_callback(GLFWwindow *window, int width, int height) {
  glViewport(0, 0, width, height);
}

void process_input(GLFWwindow *window) {
  if (glfwGetKey(window, GLFW_KEY_ESCAPE) == GLFW_PRESS) {
    glfwSetWindowShouldClose(window, true);
  }
}
